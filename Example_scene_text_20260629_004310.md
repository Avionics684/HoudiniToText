```c
// タイル側の各 connected island に対して、
// 一番近い母材primitiveの name をコピーする。
// これで「タイル」と「裏の母材」が同じRBD pieceになる。

if (!hasprimattrib(0, "tile_id"))
{
    warning("tile_id がありません。polyextrude後に Connectivity SOP を入れてください。");
    return;
}

if (!hasprimattrib(1, "name"))
{
    warning("Input 1 に primitive attribute 'name' がありません。母材側の name を確認してください。");
    return;
}

int ntile = nuniqueval(0, "prim", "tile_id");

for (int i = 0; i < ntile; i++)
{
    int id = uniqueval(0, "prim", "tile_id", i);
    int count = findattribvalcount(0, "prim", "tile_id", id);

    vector center = {0, 0, 0};
    int npts_total = 0;
    int prims[];

    // タイル1枚ぶんの中心を求める
    for (int j = 0; j < count; j++)
    {
        int pr = findattribval(0, "prim", "tile_id", id, j);
        append(prims, pr);

        int pts[] = primpoints(0, pr);
        foreach (int pt; pts)
        {
            center += point(0, "P", pt);
            npts_total++;
        }
    }

    if (npts_total <= 0)
        continue;

    center /= float(npts_total);

    // 母材側で一番近いprimitiveを探す
    int hitprim;
    vector uvw;
    float dist = xyzdist(1, center, hitprim, uvw);

    if (hitprim < 0)
        continue;

    string backing_name = prim(1, "name", hitprim);

    // タイル1枚全体に、母材と同じ name を付ける
    foreach (int pr; prims)
    {
        setprimattrib(0, "name", pr, backing_name, "set");
    }
}
```
